\newHymn
\FirstLine{Pange, lingua, gloriosi}


\begin{verse}[\versewidth]

\marginnote{\footnotesize Sing my tongue the Saviour's glory
	Of His flesh the mystery sing;
Of the Blood, all price exceeding,
Shed by our immortal King,
Destined, for the world's redemption, 
From a noble womb to spring.}
\FirstVerse{P}{ange,} lingua, gloriósi\\
Córporis mystérium,    \\
Sanguinísque pretiósi,   \\  
Quem in mundi prétium    \\
Fructus ventris generósi   \\
Rex effúdit Géntium.        
%\pointtrans                            

\marginnote{\footnotesize Of a pure and spotless Virgin
Born for us on earth below,
He, as Man, with man conversing,
Stayed, the seeds of truth to sow;  
Then He closed in solemn order
Wondrously His life of woe.}
\flagverse{2}Nobis datus, nobis natus   \\
Ex intácta Vírgine,        \\
Et in mundo conversátus,   \\
Sparso verbi sémine,       \\
Sui moras incolátus        \\
Miro clausit órdine.
                  
\marginnote{\footnotesize On the night of that Last Supper,
Seated with His chosen band,  
He the Pascal victim eating,  
First fulfills the Law's command; 
Then as Food to His Apostles  
Gives Himself with His own hand.}
\flagverse{3}In suprémae nocte coenae    \\
Recúmbens cum frátribus    \\
Observáta lege plene       \\
Cibis in legálibus,        \\
Cibum turbae duodénae      \\
Se dat suis mánibus.        

\marginnote{\footnotesize Word-made-Flesh, the bread of nature
By His word to Flesh He turns;
Wine into His Blood He changes;--- 
What though sense no change discerns? 
Only be the heart in earnest, 
Faith her lesson quickly learns.}
\flagverse{4}Verbum caro, panem verum\\   
Verbo carnem éfficit:      \\
Fitque sanguis Christi merum,\\
Et si sensus déficit,      \\
Ad firmándum cor sincérum  \\
Sola fides súfficit.        

\marginnote{\footnotesize Down in adoration falling,
Lo! the sacred Host we hail;  
Lo! o'er ancient forms departing,
Newer rites of grace prevail; 
Faith for all defects supplying,
Where the feeble senses fail.}
\flagverse{5}Tantum ergo Sacraméntum  \\  
Venerémur cérnui:          \\
Et antíquum documéntum     \\
Novo cedat rítui:          \\
Praestet fides suppleméntum\\
Sénsuum deféctui.           

\marginnote{\footnotesize To the everlasting Father,
And the Son who reigns on high,     
With the Holy Ghost proceeding
Forth from Each eternally,
Be salvation, honour, blessing,
Might and endless majesty.}
\flagverse{6}Genitóri, Genitóque\\ 
Laus et jubilátio,\\
Salus, honor, virtus quoque\\
Sit et benedíctio:  \\
Procedénti ab utróque \\     
Compar sit laudátio.\\
Amen. %Alleluia.             

\end{verse}

\Hpoet{St.~Thomas Aquinas}{1225--74}
